<%*
/* 
 * Obsidian Novel Visualization Toolkit
 * 初始化脚本 v1.0
 * 
 * 使用 Templater 在根目录创建所有必要的文件夹
 * 运行一次即可完成初始化
 */

const root = app.vault.root;
const folders = [
    "characters",
    "chapters", 
    "factions",
    "plotlines",
    "templates",
    "scripts",
    "visualization",
    "metadata"
];

let created = [];
let exists = [];

for (const folder of folders) {
    const exists_folder = await app.vault.getFolderByPathHas(folder);
    if (exists_folder) {
        exists.push(folder);
    } else {
        await app.vault.createFolder(folder);
        created.push(folder);
    }
}

// 输出结果
if (created.length > 0) {
    dv.paragraph(`✅ 已创建文件夹：${created.join(", ")}`);
}

if (exists.length > 0) {
    dv.paragraph(`📁 已存在文件夹：${exists.join(", ")}`);
}

dv.header(2, "📋 当前文件夹结构");

const allFolders = app.vault.getAllLoadedFiles()
    .filter(f => f.children);

const folderList = allFolders
    .filter(f => folders.includes(f.name))
    .map(f => [f.name, f.children.length + " 个子项"]);

if (folderList.length > 0) {
    dv.table(["文件夹", "内容"], folderList);
} else {
    dv.paragraph("*暂无文件夹数据*");
}

%>
