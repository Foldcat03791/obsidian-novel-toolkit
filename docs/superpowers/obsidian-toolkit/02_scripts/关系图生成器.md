```dataviewjs
/* 
 * Obsidian Novel Visualization Toolkit
 * 关系图自动生成器 v1.0
 * 
 * 使用方法：
 * 1. 将此代码块复制到 Obsidian 笔记中
 * 2. 确保角色笔记位于 "characters" 文件夹
 * 3. 代码将自动生成 Mermaid 关系图代码
 */

const FOLDER_CHARACTERS = "characters";
const FOLDER_FACTIONS = "factions";

// 关系类型配置
const RELATION_TYPES = {
    "ally": { label: "盟友", symbol: "--", direction: "bidirectional" },
    "enemy": { label: "敌对", symbol: "-.->", direction: "unidirectional" },
    "family": { label: "家人", symbol: "==", direction: "bidirectional" },
    "romantic": { label: "恋人", symbol: "==>", direction: "bidirectional" },
    "mentor": { label: "师徒", symbol: "-->", direction: "unidirectional" },
    "rival": { label: "竞争", symbol: "-.->>", direction: "unidirectional" },
    "neutral": { label: "中立", symbol: "...", direction: "bidirectional" }
};

// 关系强度配置
const STRENGTH_CONFIG = {
    "intense": { label: "生死之交", weight: 3 },
    "strong": { label: "深厚羁绊", weight: 2 },
    "moderate": { label: "一般", weight: 1 },
    "weak": { label: "表面", weight: 0.5 }
};

// 阵营颜色配置
const FACTION_COLORS = {
    "protagonist": "#4CAF50",
    "antagonist": "#F44336",
    "neutral": "#9E9E9E",
    "mysterious": "#9C27B0",
    "default": "#2196F3"
};

// 获取阵营颜色
function getFactionColor(faction) {
    if (!faction) return FACTION_COLORS.default;
    const lower = faction.toLowerCase();
    if (lower.includes("protagonist")) return FACTION_COLORS.protagonist;
    if (lower.includes("antagonist")) return FACTION_COLORS.antagonist;
    if (lower.includes("neutral")) return FACTION_COLORS.neutral;
    if (lower.includes("mysterious")) return FACTION_COLORS.mysterious;
    return FACTION_COLORS.default;
}

// 生成节点ID
function generateNodeId(name) {
    if (!name) return "unknown";
    return name
        .toLowerCase()
        .replace(/[^a-z0-9\u4e00-\u9fa5]/g, "-")
        .replace(/-+/g, "-")
        .substring(0, 20);
}

// 按阵营分组
function groupByFaction(characters) {
    const groups = {};
    characters.forEach(char => {
        const faction = char.faction || "default";
        if (!groups[faction]) {
            groups[faction] = [];
        }
        groups[faction].push(char);
    });
    return groups;
}

// 生成带阵营分组的Mermaid图
function generateMermaidGraphWithFactions(characters, relationships) {
    let mermaid = `flowchart TB
    %%{init: {'theme': 'base', 'themeVariables': {
        'primaryColor': '#4A90E2',
        'primaryTextColor': '#fff',
        'primaryBorderColor': '#2E5090',
        'lineColor': '#666',
        'secondaryColor': '#f5f5f5'
    }}%%

`;

    const factions = groupByFaction(characters);
    const nodeIds = new Set();

    Object.entries(factions).forEach(([faction, chars]) => {
        const factionName = faction.replace(/-/g, " ").replace(/\b\w/g, c => c.toUpperCase());
        
        mermaid += `    subgraph ${faction.replace(/[^a-zA-Z]/g, "")}["${factionName}"]\n`;

        chars.forEach(char => {
            const nodeId = generateNodeId(char.name);
            nodeIds.add(nodeId);
            const label = char.name;
            const desc = char.occupation || char.role || "";
            
            mermaid += `        ${nodeId}["${label}<br/><small>${desc}</small>"]\n`;
        });

        mermaid += `    end\n\n`;
    });

    mermaid += `    %% 关系定义\n`;

    relationships.forEach(rel => {
        const sourceId = generateNodeId(rel.source);
        const targetId = generateNodeId(rel.target);
        
        if (!nodeIds.has(sourceId) || !nodeIds.has(targetId)) return;

        const relType = RELATION_TYPES[rel.type] || RELATION_TYPES.neutral;
        const strength = STRENGTH_CONFIG[rel.strength] || STRENGTH_CONFIG.moderate;
        
        let line = `    ${sourceId}`;
        
        if (relType.direction === "bidirectional") {
            line += " <";
            line += relType.symbol.substring(1);
            line += ">";
        } else {
            line += " " + relType.symbol;
        }
        
        line += `|"${strength.label}"| ${targetId}`;
        
        mermaid += line + "\n";
    });

    return mermaid;
}

// 查询角色数据
const characters = dv.pages(`"${FOLDER_CHARACTERS}"`)
    .where(p => p.name);

const relationships = [];

characters.forEach(char => {
    if (char.relations) {
        char.relations.forEach(rel => {
            relationships.push({
                source: char.name,
                target: rel.target,
                type: rel.type || "neutral",
                strength: rel.strength || "moderate",
                note: rel.note || ""
            });
        });
    }
});

// 输出统计信息
dv.header(2, "📊 角色关系统计");
dv.table(
    ["指标", "数值"],
    [
        ["总角色数", characters.length],
        ["总关系数", relationships.length]
    ]
);

// 生成Mermaid代码
const mermaidCode = generateMermaidGraphWithFactions(characters.array(), relationships);

dv.header(2, "🕸️ 全局关系图");

// 输出可复制代码
dv.paragraph("> **使用方法**：复制下方代码到 Mermaid 代码块中");

// 创建可复制的代码块
dv.paragraph("```mermaid\n" + mermaidCode + "\n```");

// 输出纯文本版本供复制
dv.header(3, "📋 可复制代码");
dv.paragraph("```\n" + mermaidCode + "\n```");
```
