```dataviewjs
/* 
 * Obsidian Novel Visualization Toolkit
 * 时间线自动生成器 v1.0
 * 
 * 使用方法：
 * 1. 将此代码块复制到 Obsidian 笔记中
 * 2. 确保章节笔记位于 "chapters" 文件夹
 * 3. 代码将自动生成 Timeline 格式代码
 */

const FOLDER_CHAPTERS = "chapters";

// 事件重要性配置
const SIGNIFICANCE_CONFIG = {
    "major": { label: "⭐⭐⭐ 重大事件", icon: "🔥" },
    "important": { label: "⭐⭐ 重要事件", icon: "⚡" },
    "minor": { label: "⭐ 一般事件", icon: "📌" }
};

// 生成Timeline格式
function generateTimelineEvents(chapters) {
    let timeline = `type: timeline
subtitle: 《你的小说名》关键事件时间线

`;
    
    // 按卷分组
    const volumes = {};
    
    chapters.forEach(ch => {
        const volume = ch.volume || "volume-01";
        if (!volumes[volume]) {
            volumes[volume] = [];
        }
        volumes[volume].push(ch);
    });
    
    // 排序
    Object.values(volumes).forEach(v => {
        v.sort((a, b) => (a.chapter || 0) - (b.chapter || 0));
    });
    
    // 生成时间线
    Object.entries(volumes).forEach(([volume, chapters]) => {
        const volumeName = volume.replace("volume-", "第").replace("-", "卷");
        timeline += `### ${volumeName}\n\n`;
        
        chapters.forEach(ch => {
            const chapter = ch.chapter || "?";
            const title = ch.title || "";
            
            timeline += `- date: 第${chapter}章\n`;
            timeline += `  content: |\n`;
            timeline += `    **${title}**\n`;
            
            if (ch.timelineEvents && ch.timelineEvents.length > 0) {
                ch.timelineEvents.forEach(event => {
                    const sig = SIGNIFICANCE_CONFIG[event.significance] || SIGNIFICANCE_CONFIG.minor;
                    timeline += `    ${sig.icon} ${event.event}\n`;
                });
            } else {
                timeline += `    - 章节进行中\n`;
            }
            
            timeline += `\n`;
        });
    });
    
    return timeline;
}

// 查询章节数据
const chapters = dv.pages(`"${FOLDER_CHAPTERS}"`)
    .where(p => p.chapter)
    .sort(p => p.chapter, "asc");

// 生成事件列表
const allEvents = [];

chapters.forEach(ch => {
    if (ch.timelineEvents) {
        ch.timelineEvents.forEach(event => {
            allEvents.push({
                chapter: ch.chapter,
                title: ch.title,
                event: event.event,
                type: event.type || "一般",
                significance: event.significance || "minor",
                date: ch.date || `第${ch.chapter}章`
            });
        });
    }
});

// 输出统计
dv.header(2, "📊 时间线事件统计");
dv.table(
    ["指标", "数值"],
    [
        ["总章节数", chapters.length],
        ["总事件数", allEvents.length],
        ["重大事件", allEvents.filter(e => e.significance === "major").length],
        ["重要事件", allEvents.filter(e => e.significance === "important").length]
    ]
);

// 输出事件表格
dv.header(2, "⏰ 事件列表");

const eventTable = allEvents.map(e => {
    const sig = SIGNIFICANCE_CONFIG[e.significance] || SIGNIFICANCE_CONFIG.minor;
    return [
        e.date,
        e.event,
        e.type,
        sig.label
    ];
});

dv.table(
    ["时间点", "事件", "类型", "重要性"],
    eventTable
);

// 生成Timeline代码
dv.header(2, "📅 Timeline 代码");
const timelineCode = generateTimelineEvents(chapters.array());

// 输出可复制代码
dv.paragraph("> **使用方法**：复制下方代码到 Timeline 格式的笔记中");
dv.paragraph("```timeline\n" + timelineCode + "\n```");

// 输出纯文本版本供复制
dv.header(3, "📋 可复制代码");
dv.paragraph("```\n" + timelineCode + "\n```");

// ===== 伏笔追踪 =====
dv.header(2, "🔮 伏笔追踪");

const allForeshadows = [];

chapters.forEach(ch => {
    if (ch.foreshadows) {
        ch.foreshadows.forEach(fs => {
            allForeshadows.push({
                chapter: ch.chapter,
                title: ch.title,
                name: fs.name || fs.id,
                status: fs.status || "planted",
                expectedReveal: fs.expectedReveal || "未知"
            });
        });
    }
});

if (allForeshadows.length > 0) {
    dv.header(3, "已铺设伏笔");
    
    const fsTable = allForeshadows.map(fs => [
        `第${fs.chapter}章`,
        fs.name,
        fs.status === "planted" ? "⏳ 进行中" : 
        fs.status === "activated" ? "⚡ 已激活" :
        fs.status === "resolved" ? "✅ 已回收" : fs.status,
        fs.expectedReveal
    ]);
    
    dv.table(
        ["铺设章节", "伏笔名", "状态", "预期回收"],
        fsTable
    );
} else {
    dv.paragraph("*暂无伏笔记录*");
}
```
