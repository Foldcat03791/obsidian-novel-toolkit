<%*
/* 
 * Obsidian Novel Visualization Toolkit
 * 快速统计面板 v1.0
 * 
 * 一键生成所有可视化数据的统计概览
 */

const FOLDER_CHARACTERS = "characters";
const FOLDER_CHAPTERS = "chapters";
const FOLDER_PLOTLINES = "plotlines";
const FOLDER_FACTIONS = "factions";

// ===== 角色统计 =====
const characters = dv.pages(`"${FOLDER_CHARACTERS}"`)
    .where(p => p.name);

const aliveCount = characters.filter(p => p.status === "alive").length;
const deadCount = characters.filter(p => p.status === "dead").length;

let relationCount = 0;
const relationTypes = {};
characters.forEach(char => {
    if (char.relations) {
        char.relations.forEach(rel => {
            relationCount++;
            relationTypes[rel.type] = (relationTypes[rel.type] || 0) + 1;
        });
    }
});

// ===== 章节统计 =====
const chapters = dv.pages(`"${FOLDER_CHAPTERS}"`)
    .where(p => p.chapter)
    .sort(p => p.chapter, "asc");

let eventCount = 0;
let foreshadowCount = 0;
chapters.forEach(ch => {
    if (ch.timeline-events) {
        eventCount += ch.timeline-events.length;
    }
    if (ch.foreshadows) {
        foreshadowCount += ch.foreshadows.length;
    }
});

// ===== 伏笔状态统计 =====
let plantedCount = 0;
let activatedCount = 0;
let resolvedCount = 0;
chapters.forEach(ch => {
    if (ch.foreshadows) {
        ch.foreshadows.forEach(fs => {
            if (fs.status === "planted") plantedCount++;
            else if (fs.status === "activated") activatedCount++;
            else if (fs.status === "resolved") resolvedCount++;
        });
    }
});

// ===== 情节线统计 =====
const plotlines = dv.pages(`"${FOLDER_PLOTLINES}"`)
    .where(p => p.name);

const activePlotlines = plotlines.filter(p => p.status === "active").length;

// ===== 输出统计面板 =====
dv.header(1, "📊 小说数据统计面板");

dv.header(2, "👥 角色统计");
dv.table(
    ["指标", "数值", "说明"],
    [
        ["总角色数", characters.length, "笔记库中的角色数量"],
        ["存活角色", aliveCount, "状态为 alive"],
        ["死亡/退出角色", deadCount, "状态为 dead"],
        ["总关系数", relationCount, "所有定义的关系"],
        ["活跃情节线", activePlotlines, "状态为 active"]
    ]
);

dv.header(2, "📖 章节统计");
dv.table(
    ["指标", "数值", "说明"],
    [
        ["总章节数", chapters.length, "已完成或进行中的章节"],
        ["关键事件", eventCount, "定义的关键事件总数"],
        ["伏笔总数", foreshadowCount, "已铺设的伏笔总数"],
        ["伏笔-已铺设", plantedCount, "状态为 planted"],
        ["伏笔-已激活", activatedCount, "状态为 activated"],
        ["伏笔-已回收", resolvedCount, "状态为 resolved"]
    ]
);

dv.header(2, "🔗 关系类型分布");
const relationTypeTable = Object.entries(relationTypes).map(([type, count]) => [
    type,
    count,
    `${((count / relationCount) * 100).toFixed(1)}%`
]);
dv.table(
    ["关系类型", "数量", "占比"],
    relationTypeTable
);

dv.header(2, "📈 章节进度");
if (chapters.length > 0) {
    const firstChapter = chapters.first().chapter;
    const lastChapter = chapters.last().chapter;
    dv.paragraph(`章节范围：第 ${firstChapter} 章 - 第 ${lastChapter} 章`);
    
    dv.progress(chapters.where(p => p.chapter <= Math.floor(lastChapter * 0.7)).length / chapters.length, 
        `已完成 ${((chapters.where(p => p.chapter <= Math.floor(lastChapter * 0.7)).length / chapters.length) * 100).toFixed(0)}%`);
} else {
    dv.paragraph("*暂无章节数据*");
}

%>
