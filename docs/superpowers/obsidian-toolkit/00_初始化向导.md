<%*
/* 
 * Obsidian Novel Visualization Toolkit
 * 初始化设置向导 v1.0
 * 
 * 第一次使用时运行此脚本
 * 自动创建所有必要的文件夹并复制模板
 */

const root = app.vault.root;
const folders = [
    { name: "characters", desc: "角色笔记" },
    { name: "chapters", desc: "章节笔记" },
    { name: "factions", desc: "阵营笔记" },
    { name: "plotlines", desc: "情节线笔记" },
    { name: "templates", desc: "模板文件夹" },
    { name: "scripts", desc: "脚本文件夹" },
    { name: "visualization", desc: "可视化文件夹" },
    { name: "metadata", desc: "元数据文件夹" }
];

let result = [];

// 创建文件夹
for (const folder of folders) {
    const exists = app.vault.getFolderByPathHas(folder.name);
    if (!exists) {
        await app.vault.createFolder(folder.name);
        result.push(`✅ 创建: ${folder.name}/ (${folder.desc})`);
    } else {
        result.push(`📁 已存在: ${folder.name}/`);
    }
}

// 输出结果
dv.header(1, "🎉 Obsidian 小说可视化工具包 - 初始化向导");

dv.header(2, "📁 文件夹创建结果");
result.forEach(msg => dv.paragraph(msg));

dv.header(2, "📋 下一步操作");
%>
